# 安装黑暗主题
~
❯ sudo pacman -S gnome-themes-extra

~
❯ cat .gtkrc
gtk-icon-theme-name = "Adwaita-dark"
gtk-theme-name = "Adwaita-dark"
gtk-font-name = "Cantarell 11"

~
❯ cat .config/gtk-3.0/settings.ini
[Settings]
gtk-icon-theme-name = Adwaita-dark
gtk-theme-name = Adwaita-dark
gtk-application-prefer-dark-theme = true
gtk-font-name = Cantarell 11
